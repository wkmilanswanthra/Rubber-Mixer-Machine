static uint8 Int_ToASCI(sint64 num, uint8* str, uint8 len, uint8 base)
{

sint64 sum = num;
	
uint8 i = 0;
	
uint8 digit;
	
uint8 Negative=0;
	

if(sum < 0){
		
	 	sum=-sum;
		
		Negative=1;
	}
	
	
do
	
{
		
	digit = sum % base;
		
	if (digit < 0xA)
			
	str[i++] = '0' + digit;
		
	else
			
	str[i++] = 'A' + digit - 0xA;
		
	sum /= base;
	}
while (sum > 0);

	
if(Negative) str[i]='-';
	
++i;
	
str[i] = '\0';
	
strrev(str);
	
}
